function mkRng(seed){let s=seed;return()=>{s^=s<<13;s^=s>>17;s^=s<<5;return(s>>>0)/4294967296;};}
function mkRandn(rng){return()=>{let u=0,v=0;while(!u)u=rng();while(!v)v=rng();return Math.sqrt(-2*Math.log(u))*Math.cos(2*Math.PI*v);};}

function generateMarketData(asset,n=500){
const rng=mkRng(ASSET_SEEDS[asset]||42);
const randn=mkRandn(rng);
const schedule=SCHEDULES[asset]||SCHEDULES.BTC;
let price=asset===“BTC”?100:asset===“ETH”?50:200;
const prices=[price];
let day=0;
for(const r of schedule){for(let i=0;i<r.l&&day<n+100;i++,day++){price*=(1+r.d+r.v*randn());prices.push(Math.max(0.01,price));}}
while(day<n+100){price*=(1+0.0005+0.015*randn());prices.push(Math.max(0.01,price));day++;}
const WV=60,WM=20;
const start=new Date(“2020-01-01”);
const data=[];
for(let i=WV;i<prices.length&&data.length<n;i++){
const ret=(prices[i]-prices[i-1])/prices[i-1];
const slice=[];
for(let j=i-WV;j<i;j++)slice.push((prices[j+1]-prices[j])/prices[j]);
const mu=slice.reduce((a,b)=>a+b,0)/WV;
const vari=slice.reduce((a,b)=>a+(b-mu)**2,0)/WV;
const std=Math.sqrt(vari)||0.001;
const vol=std*Math.sqrt(252);
const skew=slice.reduce((a,b)=>a+((b-mu)/std)**3,0)/WV;
const kurt=slice.reduce((a,b)=>a+((b-mu)/std)**4,0)/WV-3;
const maxP=Math.max(…prices.slice(Math.max(0,i-90),i+1));
const dd=(prices[i]-maxP)/maxP;
const mom20=i>=WM?(prices[i]-prices[i-WM])/prices[i-WM]:0;
const dt=new Date(start.getTime()+(i-WV)*86400000);
data.push({date:dt.toISOString().split(“T”)[0],price:prices[i],return:ret,vol60:Math.max(0.02,Math.min(2,vol)),drawdown:dd,mom20,skewness:isNaN(skew)?0:Math.max(-3,Math.min(3,skew)),kurtosis:isNaN(kurt)?0:Math.max(-3,Math.min(10,kurt)),absReturn:Math.abs(ret)});
}
return data;
}

function normalize(features){
const n=features[0].length;
const means=Array(n).fill(0),stds=Array(n).fill(0);
features.forEach(f=>f.forEach((v,i)=>{means[i]+=v;}));
means.forEach((*,i)=>{means[i]/=features.length;});
features.forEach(f=>f.forEach((v,i)=>{stds[i]+=(v-means[i])**2;}));
stds.forEach((*,i)=>{stds[i]=Math.sqrt(stds[i]/features.length)||1;});
return features.map(f=>f.map((v,i)=>(v-means[i])/stds[i]));
}

function kmeans(features,k,maxIter=150){
const norm=normalize(features);
const step=Math.floor(norm.length/k);
let centers=Array.from({length:k},(*,i)=>[…norm[i*step]]);
let labels=new Array(norm.length).fill(0);
for(let iter=0;iter<maxIter;iter++){
const nl=norm.map(f=>{let best=0,bestD=Infinity;centers.forEach((c,ci)=>{const d=Math.sqrt(f.reduce((s,v,i)=>s+(v-c[i])**2,0));if(d<bestD){bestD=d;best=ci;}});return best;});
centers=Array.from({length:k},(*,c)=>{const pts=norm.filter((*,i)=>nl[i]===c);if(!pts.length)return centers[c];return pts[0].map((*,j)=>pts.reduce((s,p)=>s+p[j],0)/pts.length);});
if(nl.every((l,i)=>l===labels[i]))break;
labels=nl;
}
const inertia=norm.reduce((s,f,i)=>{const c=centers[labels[i]];return s+f.reduce((ss,v,j)=>ss+(v-c[j])**2,0);},0);
const rs=Array.from({length:k},(*,c)=>{const pts=features.filter((*,i)=>labels[i]===c);return{c,avgVol:pts.length?pts.reduce((s,p)=>s+p[1],0)/pts.length:0};}).sort((a,b)=>a.avgVol-b.avgVol);
const remap={};rs.forEach((s,i)=>{remap[s.c]=i;});
return{labels:labels.map(l=>remap[l]),inertia};
}

function computeElbow(features,maxK=8){
return Array.from({length:maxK-1},(_,i)=>{const k=i+2;const{inertia}=kmeans(features,k,80);return{k,inertia};});
}

function computeRegimeStats(data,k){
const result=[];
for(let c=0;c<k;c++){
const pts=data.filter(d=>d.regime===c);
if(!pts.length)continue;
const rets=pts.map(d=>d.return);
const n=rets.length;
const mean=rets.reduce((a,b)=>a+b,0)/n;
const variance=rets.reduce((a,b)=>a+(b-mean)**2,0)/n;
const std=Math.sqrt(variance)||0.001;
const annRet=mean*252,annVol=std*Math.sqrt(252);
const sharpe=annRet/annVol;
const dr=rets.filter(r=>r<0);
const dStd=dr.length?Math.sqrt(dr.reduce((a,b)=>a+b**2,0)/dr.length)*Math.sqrt(252):0.001;
const sortino=annRet/dStd;
const sorted=[…rets].sort((a,b)=>a-b);
const var95=sorted[Math.floor(n*0.05)]||0;
const cvar95=sorted.slice(0,Math.floor(n*0.05)).reduce((a,b)=>a+b,0)/(Math.floor(n*0.05)||1);
const wins=rets.filter(r=>r>0),losses=rets.filter(r=>r<0);
const winRate=wins.length/n;
const avgWin=wins.length?wins.reduce((a,b)=>a+b,0)/wins.length:0;
const avgLoss=losses.length?losses.reduce((a,b)=>a+b,0)/losses.length:0;
const maxDD=Math.min(…pts.map(d=>d.drawdown));
const avgVol=pts.reduce((a,d)=>a+d.vol60,0)/n;
const calmar=maxDD?annRet/Math.abs(maxDD):0;
const skew=pts.reduce((a,d)=>a+d.skewness,0)/n;
const pf=avgLoss?Math.abs(avgWin/avgLoss):99;
result.push({c,count:n,pct:(n/data.length*100).toFixed(1),annRet:(annRet*100).toFixed(2),annVol:(annVol*100).toFixed(2),sharpe:sharpe.toFixed(3),sortino:sortino.toFixed(3),calmar:calmar.toFixed(3),var95:(var95*100).toFixed(3),cvar95:(cvar95*100).toFixed(3),winRate:(winRate*100).toFixed(1),avgWin:(avgWin*100).toFixed(3),avgLoss:(avgLoss*100).toFixed(3),maxDD:(maxDD*100).toFixed(2),avgVol:(avgVol*100).toFixed(1),skew:skew.toFixed(3),profitFactor:pf>=99?“inf”:pf.toFixed(3)});
}
return result;
}

function computeTransitionMatrix(labels,k){
const mat=Array.from({length:k},()=>Array(k).fill(0));
for(let i=1;i<labels.length;i++)mat[labels[i-1]][labels[i]]++;
return mat.map(row=>{const sum=row.reduce((a,b)=>a+b,0)||1;return row.map(v=>v/sum);});
}

function computeCumReturns(data){

return data.map(d=>{cum*=(1+d.return);return{date:d.date,cum:(cum-1)*100,regime:d.regime};});
}

function computeBacktest(data,activeRegimes){

}

function computeDurations(data,k){
const dur=Array.from({length:k},()=>[]);
let st=0;
for(let i=1;i<=data.length;i++){if(i===data.length||data[i]?.regime!==data[st].regime){dur[data[st].regime].push(i-st);st=i;}}
return dur.map((d,c)=>({c,count:d.length,avg:d.length?(d.reduce((a,b)=>a+b,0)/d.length).toFixed(1):“0”,max:d.length?Math.max(…d):0,min:d.length?Math.min(…d):0}));
}

function corrMatrix(data){
const feats=[data.map(d=>d.return),data.map(d=>d.vol60),data.map(d=>d.drawdown),data.map(d=>d.mom20),data.map(d=>d.skewness),data.map(d=>d.kurtosis)];
const means=feats.map(f=>f.reduce((a,b)=>a+b,0)/f.length);
const stds=feats.map((f,i)=>Math.sqrt(f.reduce((a,b)=>a+(b-means[i])**2,0)/f.length)||1);
return feats.map((fi,i)=>feats.map((fj,j)=>{const cov=fi.reduce((s,v,k)=>s+(v-means[i])*(fj[k]-means[j]),0)/fi.length;return cov/(stds[i]*stds[j]);}));
}
